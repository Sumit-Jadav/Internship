select t.* , u.username from user_tweets t inner join users u on u.id = t.user_id AND user_id in (select following_id from user_follows where follower_id = 1) UNION ALL select t.*,u.username from user_tweets t inner join tweet_retweets rt INNER join users u on rt.is_deleted = 0 AND rt.user_id in (select following_id from user_follows where follower_id = 1) AND rt.tweet_id = t.id AND u.id = rt.user_id;
