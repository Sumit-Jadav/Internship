select t.* , u.username from user_tweets t inner join users u on u.id = t.user_id AND user_id in (select following_id from user_follows where follower_id = 1) UNION ALL select t.*,u.username from user_tweets t inner join tweet_retweets rt INNER join users u on rt.is_deleted = 0 AND rt.user_id in (select following_id from user_follows where follower_id = 1) AND rt.tweet_id = t.id AND u.id = rt.user_id;

select t.id , t.tweet_content , t.tweet_image , t.created_at as activity_time , u.username as author , null as retweeted_by from user_tweets t join users u on u.id = t.user_id where t.user_id = 1 OR t.user_id in (select following_id from user_follows where follower_id 1) UNION ALL 
select t.id , t.tweet_content , t.tweet_image , rt.created_at as activity_time , u.username as author , ru.username as retweeted_by from tweet_retweets rt join user_tweets t on t.is = rt.tweet_id join users u on u.id = t.user_id join users ru on ru.id = rt.user_id where rt.is_deleted = 0 AND rt.user_id in (SELECT following_id from user_follows where follower_id = 1) ORDER BY activity_time desc;
